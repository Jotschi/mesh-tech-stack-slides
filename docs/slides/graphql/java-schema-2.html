private static GraphQLObjectType createQueryType() {
	return newObject().name("QueryType")

		// .demo
		.field(newFieldDefinition().name("demo")
			.type(createDemoType())
			.dataFetcher((env) -> {
				RootElement root = env.getSource();
				return root.getDemo();
			}))

		.build();
}
